Example 3

Code with Class

class Shape {
  constructor (id, x, y) {
    this.id = id;
    this.move(x, y);
  }
  move (x, y) {
    this.x = x;
    this.y = y;
  }
}

Example 3

Code with Class

class Rectangle extends Shape {
  constructor (id, x, y, width, height) {
    super(id, x, y);
    this.width  = width;
    this.height = height;
  }
}

Example 4

Generate awesomeness

let fibonacci = function* (numbers) {
  let pre = 0, cur = 1;
  while (numbers-- > 0) {
    [ pre, cur ] = [ cur, pre + cur ];
    yield cur;
  }
};

Example 4

Generate awesomeness
